001    package com.thetransactioncompany.cors;
002    
003    
004    /**
005     * CORS origin denied (not allowed) exception.
006     *
007     * @author Vladimir Dzhuvinov
008     * @version 1.5 (2012-10-19)
009     */
010    public class CORSOriginDeniedException extends CORSException {
011    
012            
013            /**
014             * The request origin.
015             */
016            private final Origin requestOrigin;
017            
018            
019            /**
020             * Creates a new CORS origin denied exception with the specified 
021             * message.
022             *
023             * @param message The message.
024             */
025            public CORSOriginDeniedException(final String message) {
026            
027                    this(message, null);
028            }
029            
030            
031            /**
032             * Creates a new CORS origin denied exception with the specified 
033             * message and request origins.
034             *
035             * @param message       The message.
036             * @param requestOrigin The request origin, {@code null} if unknown.
037             */
038            public CORSOriginDeniedException(final String message, final Origin requestOrigin) {
039            
040                    super(message);
041                    
042                    this.requestOrigin = requestOrigin;
043            }
044            
045            
046            /**
047             * Gets the request origin.
048             *
049             * @return The request origin, {@code null} if unknown or not set.
050             */
051            public Origin getRequestOrigin() {
052            
053                    return requestOrigin;
054            }
055    }
